DSS = "TOOLS\disas11.exe"

PROJECT = $(notdir $(CURDIR))
SRC = $(wildcard *.mac)
BDIR = build
#OBJ = $(patsubst %.mac, $(BDIR)/%.o, $(SRC))
OBJ = $(patsubst %.mac,%.o, $(SRC))
LST = $(patsubst %.mac,%.lst, $(SRC))

NAME= $(patsubst %.mac,%, $(SRC))

CORE = $(patsubst %.mac,%.core, $(SRC))
DISI = $(patsubst %.mac,%.disi, $(SRC))
DIS = $(patsubst %.mac,%.dis, $(SRC))

.PHONY: clean all

clean:
		rm -f *.o *.lst *.obj *.hex *.disi *.dis *.core *.bin
all:	 $(BIN) $(HEX)

%.o:	%.mac makefile
		@echo "Visi skaiciai faile $< yra oktaliniai"
		$(ASM) -o $@ -l $*.lst $<
%.bin:	%.o
	$(O2B) $< $@
	@echo "DONE BIN: $@"
	
%.core: %.bin
		$(COR) $< $*.core
		
%.disi:	%.core
		$(DSS) $< > $*.disi
		
%.dis:	%.disi
		$(CLN) $< $*.dis
.PRECIOUS: %.lst

%.hex:	%.dis
		$(OHX)  $< $*.hex
